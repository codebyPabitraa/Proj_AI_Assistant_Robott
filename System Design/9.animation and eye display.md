# Module 09 — Animation & Eye Expression Display

**Role:** Real-time facial expressions, emotional feedback, gaze control.
**Hardware:** 2x 128×64 OLED displays (eyes), servo motors for eye rotation
**Dependencies:** Pillow, Adafruit SSD1306, GPIO servo control

---

## Tech Stack

| Component | Tech | Why |
|---|---|---|
| Display | 128×64 OLED (I2C) | Low power, fast refresh, high contrast |
| Eye Servo | Micro servo 9g | Smooth gaze tracking, compact |
| Rendering | Pillow + NumPy | GPU-free, lightweight graphics |
| Animation | Custom tweens | Smooth transitions (ease-in/out) |
| State Machine | Enum + FSM | Coordinated expressions |

---

## Architecture

```
Robot State
  ├─ listening      → eyes wide, pulse
  ├─ thinking       → eyes narrowed, blink
  ├─ speaking       → mouth/eye sync
  ├─ happy          → smile + eyes happy
  ├─ confused       → raised brows
  ├─ error          → X eyes
  └─ sleeping       → closed eyes

Voice Activity
  ├─ On wake word   → pupils dilate
  ├─ During STT     → tracking sound source
  ├─ During LLM     → thinking animation
  └─ During TTS     → lip-sync if mouth present

Navigation
  ├─ Forward motion → eyes forward, ready
  ├─ Turning        → look toward turn direction
  └─ Obstacle       → eyes dart toward obstacle
```

### eye_display.py
```python
import board
import busio
import adafruit_ssd1306
from PIL import Image, ImageDraw
import numpy as np
from typing import Tuple

class EyeDisplay:
    def __init__(self, i2c_address_left: int = 0x3C, i2c_address_right: int = 0x3D):
        """
        Control dual 128×64 OLED displays for eyes.
        Each display is 128 pixels wide × 64 pixels tall.
        """
        self.width = 128
        self.height = 64

        # I2C init
        i2c = busio.I2C(board.SCL, board.SDA)

        self.display_left = adafruit_ssd1306.Adafruit_SSD1306_I2C(
            self.width, self.height, i2c, addr=i2c_address_left
        )
        self.display_right = adafruit_ssd1306.Adafruit_SSD1306_I2C(
            self.width, self.height, i2c, addr=i2c_address_right
        )

        # Create blank images
        self.image_left = Image.new("1", (self.width, self.height), 0)
        self.image_right = Image.new("1", (self.width, self.height), 0)

    def render_eye(self, image: Image, pupil_x: float, pupil_y: float, 
                   eyelid_open: float = 1.0, expression: str = "neutral"):
        """
        Render eye on image (128×64).
        
        pupil_x, pupil_y: position normalized to 0-1
        eyelid_open: 0 (closed) to 1 (open)
        expression: "neutral", "happy", "sad", "confused"
        """
        draw = ImageDraw.Draw(image)

        # Clear
        draw.rectangle([0, 0, self.width, self.height], fill=0)

        # Sclera (white of eye)
        sclera_center = (self.width // 2, self.height // 2)
        sclera_radius = 25
        draw.ellipse(
            [sclera_center[0] - sclera_radius, sclera_center[1] - sclera_radius,
             sclera_center[0] + sclera_radius, sclera_center[1] + sclera_radius],
            fill=1, outline=1
        )

        # Eyelid (top and bottom)
        eyelid_height = int(sclera_radius * (1.0 - eyelid_open))
        if eyelid_open < 1.0:
            # Top eyelid
            draw.polygon(
                [sclera_center[0] - sclera_radius, sclera_center[1] - sclera_radius,
                 sclera_center[0] + sclera_radius, sclera_center[1] - sclera_radius,
                 sclera_center[0] + sclera_radius, sclera_center[1] - sclera_radius + eyelid_height,
                 sclera_center[0] - sclera_radius, sclera_center[1] - sclera_radius + eyelid_height],
                fill=0
            )

        # Pupil
        pupil_x_px = int(sclera_center[0] + (pupil_x - 0.5) * sclera_radius)
        pupil_y_px = int(sclera_center[1] + (pupil_y - 0.5) * sclera_radius)
        pupil_radius = 8
        draw.ellipse(
            [pupil_x_px - pupil_radius, pupil_y_px - pupil_radius,
             pupil_x_px + pupil_radius, pupil_y_px + pupil_radius],
            fill=0  # Black pupil
        )

        # Highlight (reflection)
        highlight_x = pupil_x_px - 2
        highlight_y = pupil_y_px - 3
        draw.ellipse([highlight_x - 2, highlight_y - 2, highlight_x + 2, highlight_y + 2], fill=1)

        return image

    def display(self, left_params: dict, right_params: dict):
        """
        Display both eyes.
        
        left_params:  {"pupil_x": 0.5, "pupil_y": 0.5, "eyelid_open": 1.0, "expression": "neutral"}
        right_params: same
        """
        self.image_left = self.render_eye(self.image_left, **left_params)
        self.image_right = self.render_eye(self.image_right, **right_params)

        self.display_left.image(self.image_left)
        self.display_left.show()

        self.display_right.image(self.image_right)
        self.display_right.show()
```

### animations.py
```python
import time
import math
from typing import Callable

class AnimationController:
    def __init__(self, eye_display, fps: int = 30):
        self.display = eye_display
        self.fps = fps
        self.frame_time = 1.0 / fps
        self.running = False

    @staticmethod
    def ease_in_out_cubic(t: float) -> float:
        """Smooth easing function. t = 0 to 1."""
        if t < 0.5:
            return 4 * t * t * t
        else:
            f = 2 * t - 2
            return 0.5 * f * f * f + 1

    def blink(self, duration: float = 0.2, intensity: float = 1.0):
        """Blink both eyes."""
        print("[Animation] Blink")
        frames = int(duration * self.fps)
        for frame in range(frames):
            progress = frame / frames
            # Blink curve: open → close → open
            curve = -4 * progress * (progress - 1)  # Parabola
            eyelid_open = 1.0 - intensity * curve

            params_left = {
                "pupil_x": 0.5,
                "pupil_y": 0.5,
                "eyelid_open": eyelid_open,
                "expression": "neutral"
            }
            params_right = params_left.copy()

            self.display.display(params_left, params_right)
            time.sleep(self.frame_time)

    def look_at(self, azimuth: float, elevation: float = 0.0, duration: float = 0.3):
        """
        Gaze toward target angle.
        
        azimuth: -1.0 (far left) to +1.0 (far right)
        elevation: -1.0 (down) to +1.0 (up)
        """
        print(f"[Animation] Look at ({azimuth:+.1f}, {elevation:+.1f})")
        frames = int(duration * self.fps)

        start_x, start_y = 0.5, 0.5

        for frame in range(frames):
            progress = frame / frames
            smooth_progress = self.ease_in_out_cubic(progress)

            pupil_x = start_x + azimuth * 0.2 * smooth_progress
            pupil_y = start_y + elevation * 0.2 * smooth_progress

            params_left = {
                "pupil_x": pupil_x,
                "pupil_y": pupil_y,
                "eyelid_open": 1.0,
                "expression": "neutral"
            }
            params_right = params_left.copy()

            self.display.display(params_left, params_right)
            time.sleep(self.frame_time)

    def pulse_attention(self, duration: float = 1.0, intensity: float = 0.3):
        """Subtle pulse animation for attention."""
        print("[Animation] Pulse")
        frames = int(duration * self.fps)

        for frame in range(frames):
            progress = frame / frames
            # Sine wave pulse
            eyelid_open = 1.0 - intensity * 0.5 * (math.sin(progress * math.pi * 2) + 1)

            params_left = {
                "pupil_x": 0.5,
                "pupil_y": 0.5,
                "eyelid_open": eyelid_open,
                "expression": "listening"
            }
            params_right = params_left.copy()

            self.display.display(params_left, params_right)
            time.sleep(self.frame_time)

    def track_sound(self, azimuth: float, duration: float = 0.5):
        """Eyes dart toward detected sound source."""
        print(f"[Animation] Track sound at {azimuth:+.1f}°")
        # Quick look, then return to center
        self.look_at(azimuth / 180.0, 0.0, duration * 0.7)
        self.look_at(0.0, 0.0, duration * 0.3)

    def thinking_animation(self, duration: float = 2.0):
        """Complex thinking state: look up, blink slowly."""
        print("[Animation] Thinking...")
        # Look up
        self.look_at(0.0, -0.3, 0.5)
        time.sleep(0.5)
        # Slow blink
        self.blink(duration=0.3, intensity=0.8)
        # Hold closed briefly
        time.sleep(0.5)
```

```python
# Optional: Servo-controlled eye pan/tilt for more realism

import RPi.GPIO as GPIO
import time

class ServoEyeControl:
    def __init__(self, pan_pin: int = 12, tilt_pin: int = 13):
        self.pan_pin = pan_pin
        self.tilt_pin = tilt_pin
        
        GPIO.setmode(GPIO.BCM)
        GPIO.setup(self.pan_pin, GPIO.OUT)
        GPIO.setup(self.tilt_pin, GPIO.OUT)
        
        self.pan_pwm = GPIO.PWM(self.pan_pin, 50)  # 50Hz
        self.tilt_pwm = GPIO.PWM(self.tilt_pin, 50)
        self.pan_pwm.start(0)
        self.tilt_pwm.start(0)

    def set_gaze(self, pan_deg: float, tilt_deg: float):
        """
        pan_deg: -90 (left) to +90 (right)
        tilt_deg: -45 (down) to +45 (up)
        """
        # Convert degrees to PWM duty cycle (0-180° typically mapped to 2.5%-12.5%)
        pan_duty = (pan_deg + 90) / 180.0 * 10 + 2.5
        tilt_duty = (tilt_deg + 45) / 90.0 * 10 + 2.5
        
        self.pan_pwm.ChangeDutyCycle(pan_duty)
        self.tilt_pwm.ChangeDutyCycle(tilt_duty)
```

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| OLED flicker | High refresh rate + I2C lag | Buffer frames, use DMA |
| Servo jitter | PWM ripple | Use RC filter on servo pins |
| Blink timing feels robotic | Linear animation | Use easing functions (cubic) |
| I2C contention | Multiple devices on bus | Use separate I2C bus for eyes |
| Power draw too high | Both OLEDs + servos simultaneously | PWM servos only on demand |
