### camera_capture.py
```python
import cv2
import threading
from queue import Queue
import numpy as np

class CameraCapture:
    def __init__(self, device_index: int = 0, width: int = 1280, height: int = 720, fps: int = 30):
        """
        Capture video from USB camera in background thread.
        """
        self.cap = cv2.VideoCapture(device_index)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, width)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, height)
        self.cap.set(cv2.CAP_PROP_FPS, fps)

        self.frame_queue = Queue(maxsize=2)
        self.running = True

    def capture_thread(self):
        """Background thread for continuous capture."""
        while self.running:
            ret, frame = self.cap.read()
            if not ret:
                print("[Camera] Frame read failed")
                continue

            # Discard old frame if queue full
            if self.frame_queue.full():
                try:
                    self.frame_queue.get_nowait()
                except:
                    pass

            self.frame_queue.put(frame)

    def start(self):
        """Start capture thread."""
        threading.Thread(target=self.capture_thread, daemon=True).start()

    def get_frame(self) -> np.ndarray:
        """Get latest captured frame. Blocks if not available."""
        return self.frame_queue.get()

    def cleanup(self):
        self.running = False
        self.cap.release()
```

### face_recognition.py
```python
import face_recognition
import numpy as np
from pathlib import Path
from typing import List, Tuple, Optional
import os

class FaceRecognizer:
    def __init__(self, known_faces_dir: str = "data/known_faces", tolerance: float = 0.6):
        """
        Load known faces and prepare for recognition.
        
        tolerance: 0.6 is default, lower = stricter matching
        """
        self.known_face_encodings = []
        self.known_face_names = []
        self.tolerance = tolerance

        self.load_known_faces(known_faces_dir)

    def load_known_faces(self, directory: str):
        """Load and encode all known faces from directory."""
        print(f"[Face Rec] Loading known faces from {directory}")

        for person_dir in Path(directory).iterdir():
            if not person_dir.is_dir():
                continue

            person_name = person_dir.name
            print(f"  Loading {person_name}...")

            for image_file in person_dir.glob("*.jpg"):
                try:
                    image = face_recognition.load_image_file(str(image_file))
                    encodings = face_recognition.face_encodings(image)

                    if encodings:
                        self.known_face_encodings.append(encodings[0])
                        self.known_face_names.append(person_name)
                except Exception as e:
                    print(f"  Error loading {image_file}: {e}")

        print(f"[Face Rec] Loaded {len(self.known_face_names)} faces")

    def recognize_faces(self, image: np.ndarray, tolerance: Optional[float] = None) -> List[Tuple[dict, float]]:
        """
        Detect and identify all faces in image.
        
        Returns: list of (face_info, confidence)
        face_info: {"name": str, "top": int, "right": int, "bottom": int, "left": int}
        confidence: 1.0 - distance (0.0 = perfect match, 1.0 = no match)
        """
        if tolerance is None:
            tolerance = self.tolerance

        # Detect all faces in image
        face_locations = face_recognition.face_locations(image, model="hog")  # "cnn" for GPU
        face_encodings = face_recognition.face_encodings(image, face_locations)

        results = []
        for face_encoding, face_location in zip(face_encodings, face_locations):
            # Compare against known faces
            distances = face_recognition.face_distance(self.known_face_encodings, face_encoding)

            if len(distances) == 0:
                results.append(({"name": "Unknown", "location": face_location}, 0.0))
                continue

            best_match_idx = np.argmin(distances)
            best_match_distance = distances[best_match_idx]

            if best_match_distance < tolerance:
                name = self.known_face_names[best_match_idx]
                confidence = 1.0 - best_match_distance
                results.append(({"name": name, "location": face_location}, confidence))
            else:
                results.append(({"name": "Unknown", "location": face_location}, 0.0))

        return results

    def add_face_to_dataset(self, person_name: str, image: np.ndarray):
        """Add new face to known dataset (for enrollment)."""
        encodings = face_recognition.face_encodings(image)
        if encodings:
            self.known_face_encodings.append(encodings[0])
            self.known_face_names.append(person_name)
            print(f"[Face Rec] Enrolled {person_name}")
```

### object_detection.py
```python
import cv2
import numpy as np
from typing import List, Tuple
import tensorflow as tf

class YOLODetector:
    def __init__(self, model_path: str = "models/yolov8n.tflite", confidence_threshold: float = 0.5):
        """
        Initialize YOLO v8 Nano for real-time object detection.
        
        Use TFLite for faster inference on Pi.
        Download model from: https://github.com/ultralytics/ultralytics
        """
        self.confidence_threshold = confidence_threshold

        # Load TFLite interpreter
        self.interpreter = tf.lite.Interpreter(model_path=model_path)
        self.interpreter.allocate_tensors()

        self.input_details = self.interpreter.get_input_details()
        self.output_details = self.interpreter.get_output_details()

        # COCO class labels
        self.class_labels = [
            "person", "bicycle", "car", "motorbike", "aeroplane", "bus", "train", "truck",
            "boat", "traffic light", "fire hydrant", "stop sign", "parking meter", "bench",
            "cat", "dog", "horse", "sheep", "cow", "elephant", "bear", "zebra", "giraffe",
            "backpack", "umbrella", "handbag", "tie", "suitcase", "frisbee", "skis",
            # ... full COCO 80 classes
        ]

    def detect(self, image: np.ndarray) -> List[dict]:
        """
        Detect objects in image.
        
        Returns: list of {"class": str, "confidence": float, "bbox": (x, y, w, h)}
        """
        # Preprocess: resize to model input size (typically 640x640)
        input_size = self.input_details[0]["shape"][1:3]
        resized = cv2.resize(image, (input_size[1], input_size[0]))
        resized = np.expand_dims(resized, 0).astype(np.float32) / 255.0

        # Inference
        self.interpreter.set_tensor(self.input_details[0]["index"], resized)
        self.interpreter.invoke()

        # Extract detections
        output_data = self.interpreter.get_tensor(self.output_details[0]["index"])
        detections = output_data[0]  # (n_detections, 6) → [x, y, w, h, conf, class_id]

        results = []
        for detection in detections:
            x, y, w, h, conf, class_id = detection
            if conf > self.confidence_threshold:
                class_idx = int(class_id)
                class_name = self.class_labels[class_idx] if class_idx < len(self.class_labels) else "Unknown"

                # Convert to image coordinates
                h_img, w_img = image.shape[:2]
                x_px, y_px, w_px, h_px = int(x * w_img), int(y * h_img), int(w * w_img), int(h * h_img)

                results.append({
                    "class": class_name,
                    "confidence": float(conf),
                    "bbox": (x_px, y_px, w_px, h_px)
                })

        return results
```

### object_tracker.py
```python
import numpy as np
from typing import List, Dict
from scipy.spatial.distance import cdist

class SimpleObjectTracker:
    def __init__(self, max_distance: float = 50.0, max_age: int = 30):
        """
        Simple centroid tracking for multi-object tracking.
        
        max_distance: max pixels to associate detection with track
        max_age: frames to keep lost track before deletion
        """
        self.tracks: Dict[int, dict] = {}  # id → track info
        self.next_id = 0
        self.max_distance = max_distance
        self.max_age = max_age

    def update(self, detections: List[dict]) -> List[dict]:
        """
        Update tracks with new detections.
        
        Returns: list of (detection, track_id)
        """
        if not detections:
            # Age out tracks
            self.tracks = {
                tid: track for tid, track in self.tracks.items()
                if track["age"] < self.max_age
            }
            for track in self.tracks.values():
                track["age"] += 1
            return []

        # Extract centroids from detections
        current_centroids = np.array([
            [(d["bbox"][0] + d["bbox"][2]) / 2,
             (d["bbox"][1] + d["bbox"][3]) / 2]
            for d in detections
        ])

        # Extract centroids from existing tracks
        track_ids = list(self.tracks.keys())
        if track_ids:
            track_centroids = np.array([
                self.tracks[tid]["centroid"] for tid in track_ids
            ])

            # Hungarian algorithm or greedy matching
            distances = cdist(current_centroids, track_centroids)

            # Greedy matching
            matched_pairs = []
            used_detections = set()
            for track_idx, track_id in enumerate(track_ids):
                best_det_idx = np.argmin(distances[:, track_idx])
                if distances[best_det_idx, track_idx] < self.max_distance and best_det_idx not in used_detections:
                    matched_pairs.append((best_det_idx, track_id))
                    used_detections.add(best_det_idx)

            # Update matched tracks
            for det_idx, track_id in matched_pairs:
                self.tracks[track_id]["centroid"] = current_centroids[det_idx]
                self.tracks[track_id]["age"] = 0
                self.tracks[track_id]["detection"] = detections[det_idx]

            # Create new tracks for unmatched detections
            for det_idx in range(len(detections)):
                if det_idx not in used_detections:
                    self.tracks[self.next_id] = {
                        "centroid": current_centroids[det_idx],
                        "age": 0,
                        "detection": detections[det_idx]
                    }
                    self.next_id += 1
        else:
            # No existing tracks, create new ones
            for det_idx, detection in enumerate(detections):
                self.tracks[self.next_id] = {
                    "centroid": current_centroids[det_idx],
                    "age": 0,
                    "detection": detection
                }
                self.next_id += 1

        # Age out old tracks
        self.tracks = {
            tid: track for tid, track in self.tracks.items()
            if track["age"] < self.max_age
        }

        # Return all active tracks
        return [
            (track["detection"], track_id)
            for track_id, track in self.tracks.items()
        ]
```

### Main Vision Loop

```python
from modules.vision.camera_capture import CameraCapture
from modules.vision.face_recognition import FaceRecognizer
from modules.vision.object_detection import YOLODetector
from modules.vision.object_tracker import SimpleObjectTracker
import cv2

def vision_loop():
    """Main vision processing loop."""
    camera = CameraCapture(device_index=0, width=1280, height=720)
    face_rec = FaceRecognizer("data/known_faces")
    obj_detector = YOLODetector("models/yolov8n.tflite")
    tracker = SimpleObjectTracker()

    camera.start()

    while True:
        frame = camera.get_frame()

        # Detect objects
        detections = obj_detector.detect(frame)
        tracked = tracker.update(detections)

        # Detect faces
        faces = face_rec.recognize_faces(frame)

        # Visualization
        for face_info, confidence in faces:
            top, right, bottom, left = face_info["location"]
            name = face_info["name"]
            color = (0, 255, 0) if confidence > 0.6 else (0, 0, 255)

            cv2.rectangle(frame, (left, top), (right, bottom), color, 2)
            cv2.putText(frame, f"{name} ({confidence:.2f})", (left, top - 10),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, color, 2)

        # Draw object detections
        for detection, track_id in tracked:
            x, y, w, h = detection["bbox"]
            cv2.rectangle(frame, (x, y), (x + w, y + h), (255, 0, 0), 2)
            cv2.putText(frame, f"{detection['class']} {detection['confidence']:.2f}",
                       (x, y - 5), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 2)
            cv2.putText(frame, f"ID:{track_id}", (x, y + h + 20),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 0, 0), 2)

        # Display
        cv2.imshow("Vision", frame)
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

    camera.cleanup()
    cv2.destroyAllWindows()
```

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| Face recognition slow | Computing 128-dim encodings | Cache encodings, use GPU |
| Poor lighting | Face shadows/glare | Add fill light or infrared |
| YOLO misses small objects | Low resolution | Increase input image size |
| False positives | Low confidence threshold | Increase threshold |
| Multi-face confusion | Similar looking users | Add liveness detection |
| Latency > 100ms | Single-threaded processing | Use separate threads per task |
