# Module 10 — API & System Integration (FastAPI + ROS2)

**Role:** Central orchestration, inter-module communication, REST API, real-time event streaming.

## Tech Stack

| Component | Technology | Why |
|---|---|---|
| REST API | FastAPI + Uvicorn | Async, fast, built-in OpenAPI docs |
| Real-Time | WebSocket + AsyncIO | Low-latency updates to dashboard |
| ROS2 Integration | rclpy + rclcpp | Scalable multi-robot, sensor fusion |

```python
from pydantic import BaseModel, Field
from typing import Optional, List
from enum import Enum
from datetime import datetime

class RobotState(str, Enum):
    IDLE = "idle"
    LISTENING = "listening"
    THINKING = "thinking"
    SPEAKING = "speaking"
    MOVING = "moving"
    DOCKING = "docking"
    ERROR = "error"

class BatteryStatus(BaseModel):
    voltage: float = Field(..., description="Battery voltage in volts")
    current: float = Field(..., description="Current draw in amps")
    soc: float = Field(..., ge=0, le=100, description="State of charge percentage")
    temperature: float = Field(..., description="Battery temperature in Celsius")
    time_remaining: Optional[float] = Field(None, description="Estimated runtime in hours")

class RobotTelemetry(BaseModel):
    timestamp: datetime
    state: RobotState
    position_x: float = Field(..., description="X position in meters")
    position_y: float = Field(..., description="Y position in meters")
    heading: float = Field(..., description="Robot heading in degrees (-180 to 180)")
    battery: BatteryStatus
    audio_level: float = Field(..., ge=0, le=1, description="Current audio level")
    temperature_cpu: float = Field(..., description="CPU temperature in Celsius")

class NavigationGoal(BaseModel):
    target_x: float
    target_y: float
    target_heading: Optional[float] = 0.0
    priority: int = Field(default=5, ge=1, le=10)

class VoiceCommand(BaseModel):
    command: str
    confidence: float = Field(..., ge=0, le=1)
    speaker_id: Optional[int] = None
    speaker_azimuth: Optional[float] = None

class CameraFrame(BaseModel):
    timestamp: datetime
    width: int
    height: int
    format: str = "jpeg"
    data: bytes

class MeetingNote(BaseModel):
    timestamp: datetime
    speaker: str
    text: str
    duration: float
```

### api/ros2_gateway.py
```python
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import Twist, PoseStamped
from std_msgs.msg import Float32MultiArray
from typing import Optional, Callable

class ROS2Gateway(Node):
    def __init__(self):
        """Bridge between FastAPI and ROS2."""
        rclpy.init()
        super().__init__("api_gateway")

        # Publishers
        self.cmd_vel_pub = self.create_publisher(Twist, "/cmd_vel", 10)
        self.nav_goal_pub = self.create_publisher(PoseStamped, "/move_base_simple/goal", 10)

        # Subscribers
        self.odom_sub = self.create_subscription(
            PoseStamped, "/odom", self.odom_callback, 10
        )
        self.sensor_sub = self.create_subscription(
            Float32MultiArray, "/robot/sensors", self.sensor_callback, 10
        )

        self.connected = False
        self.callbacks = {}

    def connect(self):
        """Establish ROS2 connection."""
        self.connected = True
        print("[ROS2] Gateway connected")

    def disconnect(self):
        self.destroy_node()
        rclpy.shutdown()
        self.connected = False

    def is_connected(self) -> bool:
        return self.connected

    def send_cmd_vel(self, linear_x: float, angular_z: float):
        """Send velocity command."""
        twist = Twist()
        twist.linear.x = linear_x
        twist.angular.z = angular_z
        self.cmd_vel_pub.publish(twist)

    def send_navigation_goal(self, x: float, y: float, theta: float = 0.0):
        """Send navigation goal to move_base."""
        goal = PoseStamped()
        goal.header.frame_id = "map"
        goal.pose.position.x = x
        goal.pose.position.y = y
        goal.pose.orientation.z = theta
        self.nav_goal_pub.publish(goal)

    def odom_callback(self, msg: PoseStamped):
        """Handle odometry updates."""
        if "odom" in self.callbacks:
            self.callbacks["odom"](msg)

    def sensor_callback(self, msg: Float32MultiArray):
        """Handle sensor updates."""
        if "sensors" in self.callbacks:
            self.callbacks["sensors"](msg)

    def register_callback(self, topic: str, callback: Callable):
        """Register callback for topic."""
        self.callbacks[topic] = callback
```

## WebSocket Manager

```python
from typing import List, Dict
from fastapi import WebSocket

class ConnectionManager:
    def __init__(self):
        self.active_connections: Dict[str, List[WebSocket]] = {}

    async def connect(self, websocket: WebSocket, channel: str):
        await websocket.accept()
        if channel not in self.active_connections:
            self.active_connections[channel] = []
        self.active_connections[channel].append(websocket)
        print(f"[WS] Client connected to {channel}")

    def disconnect(self, websocket: WebSocket, channel: str):
        self.active_connections[channel].remove(websocket)
        print(f"[WS] Client disconnected from {channel}")

    async def broadcast(self, message: dict, channel: str):
        """Send message to all subscribers of channel."""
        if channel in self.active_connections:
            for connection in self.active_connections[channel]:
                try:
                    await connection.send_json(message)
                except Exception as e:
                    print(f"[WS] Send error: {e}")
```

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| ROS2 initialization slow | Multiple nodes | Use single Node, multiple publishers |
| WebSocket clients drop | Network timeout | Add heartbeat ping/pong |
| API latency > 100ms | AsyncIO blocking calls | Use asyncio.gather() for parallel |
| MQTT message loss | QoS=0 | Set QoS=1 for critical commands |
| CORS errors | Origin mismatch | Configure CORS properly |
