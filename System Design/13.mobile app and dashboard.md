# Module 13 — Mobile App & Dashboard

**Role:** Remote control, real-time telemetry, meeting management, robot status monitoring.
**Platforms:** iOS (Swift) + Android (Kotlin) + Web (React)
**Dependencies:** React Native / Flutter, Redux, WebSocket, Firebase

---

## Tech Stack

| Component | Technology | Why |
|---|---|---|
| Cross-Platform | React Native or Flutter | Single codebase for iOS/Android/Web |
| State Management | Redux / Riverpod | Predictable state, time-travel debug |
| Real-Time | WebSocket + AsyncIO | Live telemetry at <50ms latency |
| Local Storage | SQLite / Hive | Offline capability, caching |
| Maps | Google Maps SDK | GPS tracking, route visualization |
| Analytics | Firebase Analytics | Usage tracking, crash reporting |
| Push Notifications | FCM / APNs | Remote alerts (low battery, etc.) |

```dart
class RobotTelemetry {
  final DateTime timestamp;
  final String state;  // idle, listening, thinking, speaking, moving, docking
  final double positionX;
  final double positionY;
  final double heading;
  final BatteryStatus battery;
  final double audioLevel;
  final double cpuTemperature;

  RobotTelemetry({
    required this.timestamp,
    required this.state,
    required this.positionX,
    required this.positionY,
    required this.heading,
    required this.battery,
    required this.audioLevel,
    required this.cpuTemperature,
  });

  factory RobotTelemetry.fromJson(Map<String, dynamic> json) {
    return RobotTelemetry(
      timestamp: DateTime.parse(json['timestamp']),
      state: json['state'],
      positionX: (json['position_x'] as num).toDouble(),
      positionY: (json['position_y'] as num).toDouble(),
      heading: (json['heading'] as num).toDouble(),
      battery: BatteryStatus.fromJson(json['battery']),
      audioLevel: (json['audio_level'] as num).toDouble(),
      cpuTemperature: (json['temperature_cpu'] as num).toDouble(),
    );
  }
}
```

### providers/robot_provider.dart
```dart
import 'package:flutter/foundation.dart';
import '../models/robot.dart';
import '../services/api_service.dart';
import '../services/websocket_service.dart';

class RobotProvider extends ChangeNotifier {
  final ApiService _apiService = ApiService();
  final WebSocketService _wsService = WebSocketService();

  RobotTelemetry? _telemetry;
  List<Meeting> _meetings = [];
  String _error = '';
  bool _isConnected = false;

  RobotTelemetry? get telemetry => _telemetry;
  List<Meeting> get meetings => _meetings;
  String get error => _error;
  bool get isConnected => _isConnected;

  RobotProvider() {
    _initializeWebSocket();
  }

  Future<void> _initializeWebSocket() async {
    _wsService.onTelemetry = (data) {
      _telemetry = RobotTelemetry.fromJson(data);
      notifyListeners();
    };

    _wsService.onError = (error) {
      _error = error;
      notifyListeners();
    };

    _wsService.onConnected = () {
      _isConnected = true;
      notifyListeners();
    };

    await _wsService.connect('ws://robot.local:8000/ws/telemetry');
  }

  Future<void> navigateTo(double x, double y) async {
    try {
      await _apiService.post('/api/controls/navigate', {
        'target_x': x,
        'target_y': y,
        'target_heading': 0.0,
      });
      notifyListeners();
    } catch (e) {
      _error = e.toString();
      notifyListeners();
    }
  }

  Future<void> stopRobot() async {
    try {
      await _apiService.post('/api/controls/stop', {});
      notifyListeners();
    } catch (e) {
      _error = e.toString();
      notifyListeners();
    }
  }

  Future<void> startMeeting(String title) async {
    try {
      await _apiService.post('/api/meetings/start', {'title': title});
      notifyListeners();
    } catch (e) {
      _error = e.toString();
      notifyListeners();
    }
  }

  Future<void> endMeeting() async {
    try {
      await _apiService.post('/api/meetings/end', {});
      notifyListeners();
    } catch (e) {
      _error = e.toString();
      notifyListeners();
    }
  }

  Future<void> fetchMeetings() async {
    try {
      final response = await _apiService.get('/api/meetings');
      _meetings = (response as List)
          .map((m) => Meeting.fromJson(m))
          .toList();
      notifyListeners();
    } catch (e) {
      _error = e.toString();
      notifyListeners();
    }
  }

  Future<void> initiateDocking() async {
    try {
      await _apiService.post('/api/controls/dock', {});
      notifyListeners();
    } catch (e) {
      _error = e.toString();
      notifyListeners();
    }
  }

  @override
  void dispose() {
    _wsService.disconnect();
    super.dispose();
  }
}
```

### services/websocket_service.dart
```dart
import 'package:web_socket_channel/web_socket_channel.dart';

class WebSocketService {
  late WebSocketChannel _channel;
  Function(Map<String, dynamic>)? onTelemetry;
  Function(String)? onError;
  Function()? onConnected;

  Future<void> connect(String url) async {
    try {
      _channel = WebSocketChannel.connect(Uri.parse(url));
      onConnected?.call();

      _channel.stream.listen(
        (message) {
          try {
            final data = jsonDecode(message);
            onTelemetry?.call(data);
          } catch (e) {
            onError?.call('Parse error: $e');
          }
        },
        onError: (error) {
          onError?.call('WebSocket error: $error');
        },
        onDone: () {
          onError?.call('WebSocket disconnected');
          // Auto-reconnect after 5 seconds
          Future.delayed(Duration(seconds: 5), () => connect(url));
        },
      );
    } catch (e) {
      onError?.call('Connection error: $e');
    }
  }

  void disconnect() {
    _channel.sink.close();
  }

  void send(String message) {
    _channel.sink.add(message);
  }
}
```

### services/api_service.dart
```dart
import 'package:dio/dio.dart';
import 'package:shared_preferences/shared_preferences.dart';

class ApiService {
  final Dio _dio = Dio();
  late SharedPreferences _prefs;
  static const String _baseUrl = 'http://robot.local:8000';

  ApiService() {
    _initializeDio();
  }

  void _initializeDio() {
    _dio.options.baseUrl = _baseUrl;
    _dio.options.connectTimeout = const Duration(seconds: 10);
    _dio.options.receiveTimeout = const Duration(seconds: 10);

    // Add auth interceptor
    _dio.interceptors.add(
      InterceptorsWrapper(
        onRequest: (options, handler) async {
          _prefs = await SharedPreferences.getInstance();
          final token = _prefs.getString('auth_token');
          if (token != null) {
            options.headers['Authorization'] = 'Bearer $token';
          }
          return handler.next(options);
        },
        onError: (error, handler) {
          if (error.response?.statusCode == 401) {
            // Token expired, refresh or redirect to login
          }
          return handler.next(error);
        },
      ),
    );
  }

  Future<dynamic> get(String endpoint) async {
    try {
      final response = await _dio.get(endpoint);
      return response.data;
    } on DioException catch (e) {
      throw _handleError(e);
    }
  }

  Future<dynamic> post(String endpoint, Map<String, dynamic> data) async {
    try {
      final response = await _dio.post(endpoint, data: data);
      return response.data;
    } on DioException catch (e) {
      throw _handleError(e);
    }
  }

  String _handleError(DioException error) {
    if (error.response != null) {
      return error.response?.data['detail'] ?? 'API Error';
    }
    return error.message ?? 'Network Error';
  }
}
```

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| WebSocket disconnects | Network interruption | Auto-reconnect with backoff |
| High data usage | Live streaming camera | Reduce resolution, compression |
| Cold start slow | Large app | Lazy load screens, code splitting |
| GPS not available | Indoor environment | Use WiFi positioning (RSSI) |
| Battery drain | Constant telemetry | Adjust update frequency (1Hz → 0.5Hz) |
