## Architecture

```
[Low Battery Alert] or [Voice: "Go dock"]
  → get_dock_position()    (search for IR beacons)
  → navigate_to_dock()     (path planning to dock area)
  → align_to_dock()        (fine alignment via camera)
  → connect_charging()     (advance into dock, activate coils)
  → verify_charging()      (monitor current draw)
  → idle_until_charged()   (park at full SOC)
```

### dock_detection.py
```python
import cv2
import numpy as np
from typing import Tuple, Optional

class DockDetector:
    def __init__(self, camera_index: int = 0):
        """Detect IR dock beacons via USB camera with IR filter."""
        self.cap = cv2.VideoCapture(camera_index)
        self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, 640)
        self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, 480)

    def detect_beacons(self) -> Tuple[Optional[Tuple[int, int]], float]:
        """
        Return (center_x, center_y) of dock centroid and confidence.
        IR LEDs appear as bright spots in IR images.
        """
        ret, frame = self.cap.read()
        if not ret:
            return None, 0.0

        # Convert to HSV for better IR detection
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)

        # IR LEDs are typically high-value, low-saturation (white-ish)
        lower_ir = np.array([0, 0, 200])
        upper_ir = np.array([180, 50, 255])

        mask = cv2.inRange(hsv, lower_ir, upper_ir)

        # Morphological ops to clean noise
        kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5))
        mask = cv2.morphologyEx(mask, cv2.MORPH_CLOSE, kernel)

        # Find contours (beacons)
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)

        if not contours or len(contours) < 2:  # Need at least 2 beacons
            return None, 0.0

        # Sort by area and take top 2 beacons
        contours = sorted(contours, key=cv2.contourArea, reverse=True)[:2]
        moments = [cv2.moments(c) for c in contours]
        centroids = [(m["m10"] / m["m00"], m["m01"] / m["m00"]) for m in moments if m["m00"] != 0]

        if len(centroids) < 2:
            return None, 0.0

        # Dock center is midpoint between two beacons
        dock_x = int((centroids[0][0] + centroids[1][0]) / 2)
        dock_y = int((centroids[0][1] + centroids[1][1]) / 2)

        # Confidence based on area
        confidence = min(cv2.contourArea(contours[0]), cv2.contourArea(contours[1])) / 1000.0
        confidence = min(confidence, 1.0)

        return (dock_x, dock_y), confidence

    def release(self):
        self.cap.release()
```

### alignment_controller.py
```python
import time
import math
from modules.navigation.motor_controller import MotorController
from dock_detection import DockDetector

class DockAlignmentController:
    def __init__(self, motor_controller: MotorController):
        self.motor = motor_controller
        self.detector = DockDetector()
        self.aligned = False

    def align_to_dock(self, max_attempts: int = 30) -> bool:
        """
        Align robot to dock using visual feedback.
        Camera center (320, 240) should match dock center.
        """
        frame_width = 640
        frame_center_x = frame_width / 2

        for attempt in range(max_attempts):
            dock_pos, confidence = self.detector.detect_beacons()

            if dock_pos is None or confidence < 0.3:
                print(f"[Dock] Beacon not visible (attempt {attempt})")
                # Rotate to search
                self.motor.set_motor_speed(0.2, -0.2)
                time.sleep(0.5)
                continue

            dock_x, dock_y = dock_pos
            error_x = dock_x - frame_center_x  # Pixels from center

            print(f"[Dock] Error: {error_x:.1f}px, Confidence: {confidence:.2f}")

            if abs(error_x) < 20:  # Within 20 pixels
                print("[Dock] Aligned!")
                self.motor.set_motor_speed(0, 0)
                self.aligned = True
                return True

            # Proportional steering correction
            kp = 0.002  # Gain: convert pixel error to motor command
            u = kp * error_x
            self.motor.set_motor_speed(-u, u)  # Differential drive

            time.sleep(0.2)

        print("[Dock] Alignment failed - timeout")
        return False

    def advance_into_dock(self, distance: float = 0.3, speed: float = 0.1):
        """Move forward into dock slowly."""
        print(f"[Dock] Advancing {distance}m at {speed} m/s")
        duration = distance / speed
        start = time.time()

        while time.time() - start < duration:
            self.motor.set_motor_speed(speed, speed)
            time.sleep(0.1)

        self.motor.set_motor_speed(0, 0)
        time.sleep(0.5)  # Settle
```

### auto_dock.py
```python
import time
from typing import Optional
from modules.navigation.path_planning import PathPlanner
from modules.navigation.motor_controller import MotorController
from modules.hardware.power_management import BatteryManagementSystem
from alignment_controller import DockAlignmentController
from charging_state_machine import ChargingStateMachine

class AutoDockingSystem:
    def __init__(self, dock_location: tuple = (0, 0)):
        """
        dock_location: (x, y) in world frame where dock is located
        """
        self.dock_location = dock_location
        self.planner = PathPlanner(200, 200, cell_size=0.1)
        self.motor = MotorController(motor_pins={...})
        self.bms = BatteryManagementSystem()
        self.aligner = DockAlignmentController(self.motor)
        self.charger = ChargingStateMachine()

    def should_dock(self) -> bool:
        """Return True if robot needs to charge."""
        status = self.bms.get_status()
        return status["soc_percent"] < 20 or status["temp_c"] > 45

    def initiate_docking(self) -> bool:
        """
        Full docking sequence.
        Returns True if successfully docked and charging.
        """
        print("[Docking] Initiating autonomous dock sequence")

        # Phase 1: Navigate to dock area
        print("[Docking] Phase 1: Navigation")
        self.plan_and_drive_to_dock()
        time.sleep(1)

        # Phase 2: Coarse detection (LiDAR area search)
        print("[Docking] Phase 2: Coarse detection")
        if not self.search_for_dock():
            print("[Docking] Could not locate dock, aborting")
            return False
        time.sleep(0.5)

        # Phase 3: Fine alignment (camera-based)
        print("[Docking] Phase 3: Fine alignment")
        if not self.aligner.align_to_dock():
            print("[Docking] Alignment failed, aborting")
            return False
        time.sleep(0.5)

        # Phase 4: Advance into dock and connect
        print("[Docking] Phase 4: Advance and connect")
        self.aligner.advance_into_dock(distance=0.2, speed=0.05)
        time.sleep(1)

        # Phase 5: Activate charging
        print("[Docking] Phase 5: Activate charging")
        if not self.charger.start_charging():
            print("[Docking] Charging failed, backup and retry")
            return False

        # Phase 6: Monitor charging until full
        print("[Docking] Phase 6: Charging...")
        return self.charger.charge_until_full()

    def plan_and_drive_to_dock(self):
        """Navigate from current position to dock area."""
        lidar_points = []  # TODO: Get from LiDAR
        self.planner.update_grid(lidar_points, 0, 0)
        
        dock_grid_x = int(self.dock_location[0] / 0.1)
        dock_grid_y = int(self.dock_location[1] / 0.1)
        path = self.planner.plan((0, 0), (dock_grid_x, dock_grid_y))

        for wx, wy in path:
            self.motor.drive_to_waypoint((wx * 0.1, wy * 0.1))

    def search_for_dock(self) -> bool:
        """Spin and listen for IR beacons."""
        print("[Docking] Searching...")
        for i in range(4):  # Rotate 4x 90° = full circle
            beacon_pos, conf = self.aligner.detector.detect_beacons()
            if beacon_pos and conf > 0.5:
                print(f"[Docking] Dock found at {beacon_pos}")
                return True
            self.motor.set_motor_speed(0.2, -0.2)
            time.sleep(1)

        return False
```

import time
from enum import Enum

class ChargeState(Enum):
    IDLE = "idle"
    CHARGING = "charging"
    FAULT = "fault"
    FULL = "full"

class ChargingStateMachine:
    def __init__(self, gpio_coil_pin: int = 14, target_voltage: float = 51.0):
        """
        gpio_coil_pin: GPIO pin for inductive coil enable
        target_voltage: 48V + 3V = 51V (full for 4S LiFePO4)
        """
        self.state = ChargeState.IDLE
        self.gpio_pin = gpio_coil_pin
        self.target_voltage = target_voltage
        self.charge_start_time = None

    def start_charging(self) -> bool:
        """Enable inductive coils, verify current draw."""
        print("[Charge] Activating coils...")
        # TODO: GPIO.output(self.gpio_pin, GPIO.HIGH)
        time.sleep(0.5)

        # Check if current is flowing (indicates good alignment)
        # TODO: current = bms.read_current()
        current = 5.0  # Placeholder
        if current < 1.0:
            print("[Charge] No current detected, misaligned")
            return False

        self.state = ChargeState.CHARGING
        self.charge_start_time = time.time()
        print(f"[Charge] Charging started at {current:.1f}A")
        return True

    def charge_until_full(self, timeout_hours: int = 8) -> bool:
        """Monitor charging until battery full or timeout."""
        while True:
            # TODO: status = bms.get_status()
            voltage = 48.0  # Placeholder
            soc = 50  # Placeholder

            elapsed = time.time() - self.charge_start_time
            print(f"[Charge] {voltage:.1f}V | {soc}% | {elapsed/3600:.1f}h")

            if voltage >= self.target_voltage:
                print("[Charge] Battery full!")
                self.state = ChargeState.FULL
                # TODO: GPIO.output(self.gpio_pin, GPIO.LOW)
                return True

            if elapsed > timeout_hours * 3600:
                print("[Charge] Timeout!")
                self.state = ChargeState.FAULT
                return False

            time.sleep(300)  # Check every 5 minutes

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| Misalignment | Bearing wear | Add guide rails on dock |
| Coils not engaging | Dirt on contacts | Self-cleaning brush rings |
| Charging too slow | Coil coupling weak | Increase coil size / power |
| Robot overshoots dock | Speed too high | Reduce advance_into_dock speed |
| Camera glare | Reflective dock surface | Matte black coating on beacons |
| Timeout during charge | BMS communication lag | Add watchdog timer |
