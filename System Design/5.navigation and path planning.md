## Architecture

```
[LiDAR] 
  → lidar_scan_processor.py    (raw point cloud → occupancy grid)
  → path_planning.py           (A* on grid → waypoint list)
  → obstacle_avoidance.py      (real-time safety checks)
  → motor_controller.py        (drive motors to waypoint)
  → pick_and_place.py          (task-specific manipulation)
```

### path_planning.py
```python
import heapq
import math
from typing import List, Tuple
import numpy as np

class PathPlanner:
    def __init__(self, grid_width: int, grid_height: int, cell_size: float = 0.1):
        """
        grid_width, grid_height: cells
        cell_size: meters per cell (0.1m = 10cm grid)
        """
        self.grid_width = grid_width
        self.grid_height = grid_height
        self.cell_size = cell_size
        self.occupancy_grid = np.zeros((grid_height, grid_width))

    def update_grid(self, lidar_points: List[Tuple[float, float]], robot_x: float, robot_y: float):
        """Update occupancy grid from LiDAR points."""
        self.occupancy_grid.fill(0)  # Clear
        for px, py in lidar_points:
            gx = int((px + robot_x) / self.cell_size)
            gy = int((py + robot_y) / self.cell_size)
            if 0 <= gx < self.grid_width and 0 <= gy < self.grid_height:
                self.occupancy_grid[gy, gx] = 1  # Occupied

    def heuristic(self, pos: Tuple[int, int], goal: Tuple[int, int]) -> float:
        """Euclidean distance heuristic."""
        return math.sqrt((pos[0] - goal[0])**2 + (pos[1] - goal[1])**2)

    def neighbors(self, pos: Tuple[int, int]) -> List[Tuple[int, int]]:
        """8-neighbor connectivity."""
        x, y = pos
        dirs = [
            (x+1, y), (x-1, y), (x, y+1), (x, y-1),
            (x+1, y+1), (x-1, y-1), (x+1, y-1), (x-1, y+1)
        ]
        valid = []
        for nx, ny in dirs:
            if 0 <= nx < self.grid_width and 0 <= ny < self.grid_height:
                if self.occupancy_grid[ny, nx] == 0:  # Not occupied
                    valid.append((nx, ny))
        return valid

    def plan(self, start: Tuple[int, int], goal: Tuple[int, int]) -> List[Tuple[int, int]]:
        """A* pathfinding. Returns list of (x, y) waypoints."""
        open_set = [(0, start)]
        came_from = {}
        g_score = {start: 0}
        f_score = {start: self.heuristic(start, goal)}

        while open_set:
            _, current = heapq.heappop(open_set)

            if current == goal:
                path = []
                while current in came_from:
                    path.append(current)
                    current = came_from[current]
                path.append(start)
                return list(reversed(path))

            for neighbor in self.neighbors(current):
                tentative_g = g_score[current] + 1
                if neighbor not in g_score or tentative_g < g_score[neighbor]:
                    came_from[neighbor] = current
                    g_score[neighbor] = tentative_g
                    f_score[neighbor] = tentative_g + self.heuristic(neighbor, goal)
                    heapq.heappush(open_set, (f_score[neighbor], neighbor))

        return []  # No path found

    def waypoints_to_coords(self, waypoints: List[Tuple[int, int]]) -> List[Tuple[float, float]]:
        """Convert grid waypoints to real-world coordinates."""
        return [(wx * self.cell_size, wy * self.cell_size) for wx, wy in waypoints]
```

### obstacle_avoidance.py
```python
import numpy as np
from typing import List, Tuple

class ObstacleAvoidance:
    def __init__(self, safety_radius: float = 0.3):
        """
        safety_radius: minimum distance (meters) to keep from obstacles
        """
        self.safety_radius = safety_radius
        self.recent_readings = []

    def add_lidar_scan(self, points: List[Tuple[float, float]], max_range: float = 10.0):
        """Store LiDAR scan for obstacle detection."""
        # Filter out max_range points (likely noise)
        filtered = [p for p in points if np.sqrt(p[0]**2 + p[1]**2) < max_range]
        self.recent_readings.append(filtered)
        if len(self.recent_readings) > 5:
            self.recent_readings.pop(0)

    def get_closest_obstacle(self) -> Tuple[float, float, float]:
        """Return (distance, angle, magnitude) to closest obstacle."""
        if not self.recent_readings:
            return (float('inf'), 0, 0)

        all_points = [p for reading in self.recent_readings for p in reading]
        if not all_points:
            return (float('inf'), 0, 0)

        distances = [np.sqrt(p[0]**2 + p[1]**2) for p in all_points]
        min_idx = np.argmin(distances)
        min_dist = distances[min_idx]
        px, py = all_points[min_idx]
        angle = np.arctan2(py, px)
        return (min_dist, angle, np.linalg.norm(all_points[min_idx]))

    def is_safe_to_move(self, waypoint: Tuple[float, float]) -> bool:
        """Check if waypoint is reachable without hitting obstacles."""
        closest_dist, _, _ = self.get_closest_obstacle()
        return closest_dist > self.safety_radius

    def emergency_stop(self) -> bool:
        """Return True if collision imminent."""
        closest_dist, _, _ = self.get_closest_obstacle()
        return closest_dist < (self.safety_radius * 0.5)
```

### motor_controller.py

import time
import RPi.GPIO as GPIO
from typing import Tuple

class MotorController:
    def __init__(self, motor_pins: dict, wheel_radius: float = 0.05):
        """
        motor_pins: {"left_forward": 17, "left_backward": 27, ...}
        wheel_radius: meters
        """
        self.motor_pins = motor_pins
        self.wheel_radius = wheel_radius
        self.wheel_base = 0.15  # Distance between wheels (meters)

        GPIO.setmode(GPIO.BCM)
        for pin in motor_pins.values():
            GPIO.setup(pin, GPIO.OUT)

    def set_motor_speed(self, left_speed: float, right_speed: float):
        """
        left_speed, right_speed: -1.0 to 1.0 (PWM duty cycle)
        Negative = backward, positive = forward
        """
        # TODO: Implement PWM control via GPIO
        # For now, just set direction
        left_dir = "FWD" if left_speed > 0 else "BWD"
        right_dir = "FWD" if right_speed > 0 else "BWD"
        print(f"[Motor] L:{left_dir} R:{right_dir} | L:{left_speed:.2f} R:{right_speed:.2f}")

    def drive_to_waypoint(self, waypoint: Tuple[float, float], max_speed: float = 0.5):
        """Drive robot to waypoint using feedback control."""
        distance = (waypoint[0]**2 + waypoint[1]**2)**0.5
        angle = __import__('math').atan2(waypoint[1], waypoint[0])

        # Simple proportional control
        print(f"[Motor] Driving to ({waypoint[0]:.2f}, {waypoint[1]:.2f})")
        
        # Phase 1: Rotate to face waypoint
        kp_rotation = 0.5
        while abs(angle) > 0.1:
            u = kp_rotation * angle
            self.set_motor_speed(u * max_speed, -u * max_speed)
            time.sleep(0.1)
            angle -= 0.1  # Simulate rotation

        # Phase 2: Drive forward
        kp_distance = 1.0
        while distance > 0.05:
            u = kp_distance * distance
            self.set_motor_speed(u * max_speed, u * max_speed)
            time.sleep(0.1)
            distance -= 0.02  # Simulate movement

        self.set_motor_speed(0, 0)
        print("[Motor] Waypoint reached")

    def stop(self):
        """Stop all motors."""
        self.set_motor_speed(0, 0)

    def cleanup(self):
        """Release GPIO."""
        GPIO.cleanup()

- **Encoders:** Monitor ticks for odometry feedback.
- **Power:** Separate 12V battery for motors, 5V regulated for Pi.

## Main Loop Integration

```python
from modules.navigation.path_planning import PathPlanner
from modules.navigation.obstacle_avoidance import ObstacleAvoidance
from modules.navigation.motor_controller import MotorController

planner = PathPlanner(200, 200, cell_size=0.1)
avoidance = ObstacleAvoidance(safety_radius=0.3)
motor = MotorController(motor_pins={...})

def navigate_to_goal(goal_x: float, goal_y: float):
    # LiDAR scan (from ROS topic subscription)
    lidar_points = lidar_scan()
    planner.update_grid(lidar_points, robot_x, robot_y)
    
    # Generate path
    path = planner.plan((0, 0), (goal_x, goal_y))
    
    # Execute waypoints
    for wx, wy in path:
        if not avoidance.is_safe_to_move((wx, wy)):
            print("Obstacle detected, replanning...")
            continue
        
        if avoidance.emergency_stop():
            motor.stop()
            break
        
        motor.drive_to_waypoint((wx, wy))
```

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| Motor jerky movement | Encoder noise | Low-pass filter encoder ticks |
| Path unreachable | Grid resolution too coarse | Reduce cell_size to 0.05m |
| Frequent re-planning | Noisy LiDAR | Fuse scans over 5-10 frames |
| Gripper doesn't grip | Servo underpowered | Use 5kg+ servo, increase PWM range |
| Robot overshoots waypoint | PID gains too high | Reduce Kp, increase damping |
| LiDAR spins too slow | Firmware issue | Update RPLiDAR firmware |
