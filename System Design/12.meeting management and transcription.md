| Component | Library | Why |

### transcription.py
```python
import whisper
import numpy as np
from typing import List, Tuple
from datetime import datetime
import time

class MeetingTranscriber:
    def __init__(self, model_name: str = "base"):
        """
        Initialize Whisper for real-time meeting transcription.
        
        Models: tiny, base, small, medium, large
        base = good balance of speed/accuracy for Pi
        """
        self.model = whisper.load_model(model_name)
        self.transcript_lines: List[dict] = []
        self.start_time = datetime.now()

    def transcribe_chunk(self, audio_chunk: np.ndarray, sample_rate: int = 16000) -> str:
        """
        Transcribe audio chunk (e.g., 30 seconds).
        
        Returns: transcribed text
        """
        # Convert numpy array to WAV format for Whisper
        import tempfile
        import soundfile as sf

        with tempfile.NamedTemporaryFile(suffix=".wav", delete=False) as tmp:
            sf.write(tmp.name, audio_chunk, sample_rate)
            result = self.model.transcribe(tmp.name, language="en")

        text = result["text"].strip()
        timestamp = (datetime.now() - self.start_time).total_seconds()

        self.transcript_lines.append({
            "timestamp": timestamp,
            "text": text,
            "speaker": None  # Filled by diarization
        })

        return text

    def get_transcript(self) -> List[dict]:
        """Return all transcribed lines."""
        return self.transcript_lines

    def export_transcript_md(self) -> str:
        """Export transcript as Markdown."""
        md = f"# Meeting Transcript\n\nStarted: {self.start_time.isoformat()}\n\n"

        for line in self.transcript_lines:
            time_str = time.strftime("%M:%S", time.gmtime(line["timestamp"]))
            speaker = line["speaker"] or "Unknown"
            md += f"**{speaker}** [{time_str}]:\n{line['text']}\n\n"

        return md
```

### diarization.py
```python
import numpy as np
from scipy.spatial.distance import euclidean
from typing import List, Tuple

class SpeakerDiarizer:
    def __init__(self, n_speakers_max: int = 10):
        """
        Assign speakers to transcript lines.
        Uses speaker embeddings from audio features.
        """
        self.n_speakers_max = n_speakers_max
        self.speaker_profiles = {}  # speaker_id → embedding
        self.speaker_names = {}     # speaker_id → name (optional)
        self.next_speaker_id = 0

    def extract_speaker_embedding(self, audio_chunk: np.ndarray, sample_rate: int = 16000) -> np.ndarray:
        """
        Extract speaker embedding from audio chunk.
        
        Uses simple MFCC features (in production, use x-vectors or speaker-net).
        Returns: speaker embedding (feature vector)
        """
        from scipy.fft import fft
        from scipy import signal

        # Compute spectrogram
        freq, time_frames, spec = signal.spectrogram(
            audio_chunk, fs=sample_rate, nperseg=2048
        )

        # Average across time to get speaker signature
        embedding = np.mean(spec, axis=1)
        return embedding / (np.linalg.norm(embedding) + 1e-8)

    def assign_speaker(self, audio_chunk: np.ndarray, transcript_line: dict, 
                      threshold: float = 0.7) -> str:
        """
        Identify speaker from audio chunk.
        
        Returns: speaker_id
        """
        embedding = self.extract_speaker_embedding(audio_chunk)

        if not self.speaker_profiles:
            # First speaker
            speaker_id = self.next_speaker_id
            self.speaker_profiles[speaker_id] = embedding
            self.speaker_names[speaker_id] = f"Speaker {speaker_id}"
            self.next_speaker_id += 1
            return str(speaker_id)

        # Compare with existing speakers
        distances = {}
        for sid, profile in self.speaker_profiles.items():
            distances[sid] = euclidean(embedding, profile)

        best_match = min(distances, key=distances.get)
        best_distance = distances[best_match]

        if best_distance < threshold and len(self.speaker_profiles) < self.n_speakers_max:
            # Update speaker profile (exponential moving average)
            alpha = 0.1
            self.speaker_profiles[best_match] = (
                alpha * embedding + (1 - alpha) * self.speaker_profiles[best_match]
            )
            return str(best_match)
        else:
            # New speaker
            speaker_id = self.next_speaker_id
            self.speaker_profiles[speaker_id] = embedding
            self.speaker_names[speaker_id] = f"Speaker {speaker_id}"
            self.next_speaker_id += 1
            return str(speaker_id)

    def set_speaker_name(self, speaker_id: str, name: str):
        """Manually set speaker name (e.g., from face recognition)."""
        if int(speaker_id) in self.speaker_names:
            self.speaker_names[int(speaker_id)] = name

    def get_speaker_name(self, speaker_id: str) -> str:
        """Get speaker name."""
        return self.speaker_names.get(int(speaker_id), f"Speaker {speaker_id}")
```

### summarizer.py
```python
import spacy
from transformers import pipeline
from typing import List, Dict

class MeetingSummarizer:
    def __init__(self):
        """Initialize NLP models for summarization."""
        # Load spaCy for NER and key phrase extraction
        self.nlp = spacy.load("en_core_web_sm")

        # Load BART for abstractive summarization
        self.summarizer = pipeline(
            "summarization",
            model="facebook/bart-large-cnn",
            device=0  # GPU if available
        )

    def extract_entities(self, transcript_text: str) -> List[dict]:
        """Extract named entities (people, organizations, dates, etc)."""
        doc = self.nlp(transcript_text)
        entities = []
        for ent in doc.ents:
            entities.append({
                "text": ent.text,
                "label": ent.label_,  # PERSON, ORG, DATE, GPE, etc.
                "start": ent.start_char,
                "end": ent.end_char
            })
        return entities

    def extract_action_items(self, transcript_lines: List[dict]) -> List[str]:
        """Extract action items (decisions, TODOs)."""
        keywords = ["action", "todo", "task", "decide", "approved", "deadline", "next"]

        action_items = []
        for line in transcript_lines:
            text_lower = line["text"].lower()
            if any(keyword in text_lower for keyword in keywords):
                action_items.append(line["text"])

        return action_items

    def generate_summary(self, transcript_text: str, max_length: int = 150, 
                        min_length: int = 50) -> str:
        """Generate abstractive summary using BART."""
        if len(transcript_text.split()) < 100:
            return transcript_text  # Too short to summarize

        summary = self.summarizer(
            transcript_text,
            max_length=max_length,
            min_length=min_length,
            do_sample=False
        )
        return summary[0]["summary_text"]

    def generate_meeting_report(self, transcript_lines: List[dict], 
                                meeting_title: str = "Meeting Report") -> dict:
        """Generate comprehensive meeting report."""
        # Full transcript
        full_text = " ".join([line["text"] for line in transcript_lines])

        # Entities
        entities = self.extract_entities(full_text)

        # Action items
        action_items = self.extract_action_items(transcript_lines)

        # Summary
        summary = self.generate_summary(full_text)

        # Build report
        report = {
            "title": meeting_title,
            "entities": entities,
            "action_items": action_items,
            "summary": summary,
            "participants": list(set(line.get("speaker", "Unknown") for line in transcript_lines)),
            "duration_minutes": transcript_lines[-1]["timestamp"] / 60 if transcript_lines else 0
        }

        return report
```

### pdf_export.py
```python
from reportlab.lib.pagesizes import letter, A4
from reportlab.lib.styles import getSampleStyleSheet, ParagraphStyle
from reportlab.lib.units import inch
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Table, TableStyle, PageBreak
from reportlab.lib import colors
from datetime import datetime
from typing import Dict, List

class MeetingReportGenerator:
    def __init__(self, output_path: str):
        self.output_path = output_path
        self.styles = getSampleStyleSheet()

    def generate_pdf(self, report_data: Dict, transcript_lines: List[dict]):
        """Generate PDF report."""
        doc = SimpleDocTemplate(self.output_path, pagesize=letter)
        story = []

        # Title
        title_style = ParagraphStyle(
            "CustomTitle",
            parent=self.styles["Heading1"],
            fontSize=24,
            textColor=colors.HexColor("#1f4788"),
            spaceAfter=30,
            alignment=1  # Center
        )
        story.append(Paragraph(report_data["title"], title_style))
        story.append(Spacer(1, 0.2 * inch))

        # Metadata
        metadata = [
            [f"Date: {datetime.now().strftime('%Y-%m-%d %H:%M')}"],
            [f"Duration: {report_data['duration_minutes']:.1f} minutes"],
            [f"Participants: {', '.join(report_data['participants'])}"]
        ]
        metadata_table = Table(metadata)
        story.append(metadata_table)
        story.append(Spacer(1, 0.3 * inch))

        # Summary
        story.append(Paragraph("Summary", self.styles["Heading2"]))
        story.append(Paragraph(report_data["summary"], self.styles["Normal"]))
        story.append(Spacer(1, 0.2 * inch))

        # Action Items
        story.append(Paragraph("Action Items", self.styles["Heading2"]))
        action_items = report_data.get("action_items", [])
        for item in action_items:
            story.append(Paragraph(f"• {item}", self.styles["Normal"]))
        story.append(Spacer(1, 0.2 * inch))

        # Key Entities
        story.append(Paragraph("Key Entities", self.styles["Heading2"]))
        entities = report_data.get("entities", [])
        entity_table_data = [["Entity", "Type"]]
        for ent in entities[:10]:  # Limit to 10
            entity_table_data.append([ent["text"], ent["label"]])
        entity_table = Table(entity_table_data)
        entity_table.setStyle(TableStyle([
            ("BACKGROUND", (0, 0), (-1, 0), colors.grey),
            ("TEXTCOLOR", (0, 0), (-1, 0), colors.whitesmoke),
            ("ALIGN", (0, 0), (-1, -1), "LEFT"),
            ("FONTNAME", (0, 0), (-1, 0), "Helvetica-Bold"),
            ("FONTSIZE", (0, 0), (-1, 0), 12),
            ("BOTTOMPADDING", (0, 0), (-1, 0), 12),
            ("GRID", (0, 0), (-1, -1), 1, colors.black)
        ]))
        story.append(entity_table)

        # Full Transcript
        story.append(PageBreak())
        story.append(Paragraph("Full Transcript", self.styles["Heading2"]))
        for line in transcript_lines:
            speaker = line.get("speaker", "Unknown")
            time_str = f"{int(line['timestamp'] // 60):02d}:{int(line['timestamp'] % 60):02d}"
            story.append(Paragraph(
                f"<b>{speaker}</b> [{time_str}]:<br/>{line['text']}",
                self.styles["Normal"]
            ))
            story.append(Spacer(1, 0.1 * inch))

        # Build PDF
        doc.build(story)
        print(f"[PDF] Report saved to {self.output_path}")
```

### Meeting State Machine

```python
from enum import Enum

class MeetingState(Enum):
    IDLE = "idle"
    WAITING_START = "waiting_start"
    RECORDING = "recording"
    PAUSED = "paused"
    ENDED = "ended"

class MeetingManager:
    def __init__(self):
        self.state = MeetingState.IDLE
        self.transcriber = MeetingTranscriber()
        self.diarizer = SpeakerDiarizer()
        self.summarizer = MeetingSummarizer()

    def start_meeting(self, title: str = "Meeting"):
        """Start meeting recording."""
        print(f"[Meeting] Starting: {title}")
        self.state = MeetingState.RECORDING
        self.title = title

    def end_meeting(self) -> str:
        """End meeting and generate report."""
        print("[Meeting] Ending and generating report...")
        self.state = MeetingState.ENDED

        # Generate report
        transcript_text = " ".join([line["text"] for line in self.transcriber.get_transcript()])
        report = self.summarizer.generate_meeting_report(
            self.transcriber.get_transcript(),
            meeting_title=self.title
        )

        # Export to PDF
        pdf_gen = MeetingReportGenerator(f"data/meeting_records/{self.title}.pdf")
        pdf_gen.generate_pdf(report, self.transcriber.get_transcript())

        return report
```
