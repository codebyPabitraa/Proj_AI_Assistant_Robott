### audio_capture.py
```python
import pyaudio
import numpy as np
from threading import Thread
from queue import Queue
import struct

class MultiChannelAudioCapture:
    def __init__(self, n_channels: int = 4, sample_rate: int = 16000, chunk_size: int = 512):
        """
        Capture synchronized audio from multiple USB mics.
        Each channel represents one mic in the line array.
        """
        self.n_channels = n_channels
        self.sample_rate = sample_rate
        self.chunk_size = chunk_size
        self.audio_queue = Queue(maxsize=100)

        self.pa = pyaudio.PyAudio()
        # Assumes 4-channel USB interface (e.g., Behringer UCA-202 modified)
        self.stream = self.pa.open(
            format=pyaudio.paInt16,
            channels=n_channels,
            rate=sample_rate,
            input=True,
            frames_per_buffer=chunk_size,
            input_device_index=2  # USB device index (varies per system)
        )

    def capture_thread(self):
        """Continuous capture in background thread."""
        while True:
            try:
                data = self.stream.read(self.chunk_size, exception_on_overflow=False)
                # Convert byte stream to numpy array
                audio_data = np.frombuffer(data, dtype=np.int16)
                # Reshape to (chunk_size, n_channels)
                audio_data = audio_data.reshape((self.chunk_size, self.n_channels))
                self.audio_queue.put(audio_data)
            except Exception as e:
                print(f"[Audio] Capture error: {e}")

    def start(self):
        """Start background capture thread."""
        Thread(target=self.capture_thread, daemon=True).start()

    def get_chunk(self) -> np.ndarray:
        """Get next audio chunk. Blocks if empty."""
        return self.audio_queue.get()

    def cleanup(self):
        self.stream.stop_stream()
        self.stream.close()
        self.pa.terminate()
```

### tdoa_tracking.py
```python
import numpy as np
from scipy import signal
from typing import Tuple

class TDOATracker:
    def __init__(self, n_channels: int = 4, sample_rate: int = 16000, 
                 mic_spacing: float = 0.05):
        """
        Compute TDOA (Time Difference of Arrival) between mics.
        
        mic_spacing: distance between adjacent mics (meters), e.g., 5cm
        n_channels: number of microphones in line array
        
        Speed of sound ≈ 343 m/s at 20°C
        Max time delay ≈ 0.05m / 343 m/s ≈ 0.15ms
        At 16kHz, this is ~2.4 samples difference
        """
        self.n_channels = n_channels
        self.sample_rate = sample_rate
        self.mic_spacing = mic_spacing
        self.c = 343.0  # Speed of sound (m/s)

    def compute_gcc_phat(self, signal1: np.ndarray, signal2: np.ndarray, 
                         max_lag: int = 50) -> Tuple[np.ndarray, int]:
        """
        Generalized Cross-Correlation with Phase Transform (GCC-PHAT).
        More robust than simple cross-correlation for noisy environments.
        
        Returns: (correlation, peak_lag_in_samples)
        """
        # Compute FFT of both signals
        fft1 = np.fft.rfft(signal1, n=len(signal1) + len(signal2) - 1)
        fft2 = np.fft.rfft(signal2, n=len(signal1) + len(signal2) - 1)

        # Cross-spectrum
        cross_spectrum = fft1 * np.conj(fft2)

        # Phase transform: normalize by magnitude
        phasors = cross_spectrum / (np.abs(cross_spectrum) + 1e-8)

        # IFFT back to time domain
        correlation = np.fft.irfft(phasors)

        # Find peak within ±max_lag samples
        center = len(correlation) // 2
        search_region = correlation[center - max_lag : center + max_lag]
        peak_idx = np.argmax(search_region)
        peak_lag = peak_idx - max_lag

        return correlation, peak_lag

    def estimate_azimuth(self, audio_chunk: np.ndarray) -> float:
        """
        Estimate source azimuth from multichannel audio.
        
        audio_chunk: shape (chunk_size, n_channels)
        Returns: angle in degrees (-180 to +180)
        
        For linear array, we only get left-right bearing.
        """
        if audio_chunk.shape[1] < 2:
            return 0.0

        # Compare first two channels
        ch0 = audio_chunk[:, 0].astype(np.float32)
        ch1 = audio_chunk[:, 1].astype(np.float32)

        # Normalize
        ch0 = ch0 / (np.max(np.abs(ch0)) + 1e-8)
        ch1 = ch1 / (np.max(np.abs(ch1)) + 1e-8)

        _, peak_lag = self.compute_gcc_phat(ch0, ch1)

        # Convert sample lag to time delay
        time_delay = peak_lag / self.sample_rate  # seconds

        # Time delay to angle (Angle of Arrival)
        # sin(theta) = (c * td) / d_mic
        sin_theta = (self.c * time_delay) / self.mic_spacing
        sin_theta = np.clip(sin_theta, -1.0, 1.0)
        azimuth_rad = np.arcsin(sin_theta)
        azimuth_deg = np.degrees(azimuth_rad)

        # If peak_lag is positive, sound came from right (positive angle)
        if peak_lag < 0:
            azimuth_deg = 180 - abs(azimuth_deg)  # Flip to right side

        return azimuth_deg

    def estimate_distance(self, audio_chunk: np.ndarray, 
                         reference_level: float = 20.0) -> float:
        """
        Rough distance estimate from RMS level.
        (More accurate with proper calibration.)
        
        reference_level: RMS at 1 meter (dB, needs measurement)
        Returns: estimated distance in meters
        """
        rms = np.sqrt(np.mean(audio_chunk**2))
        rms_db = 20 * np.log10(rms + 1e-8)

        # Inverse square law: level drops 6dB per doubling distance
        distance = 10**((reference_level - rms_db) / 20.0)
        return np.clip(distance, 0.5, 5.0)  # Clamp to reasonable range
```

### beamforming.py
```python
import numpy as np

class DelayAndSumBeamformer:
    def __init__(self, n_channels: int = 4, sample_rate: int = 16000,
                 mic_spacing: float = 0.05):
        """Delay-and-sum beamformer for focus on target angle."""
        self.n_channels = n_channels
        self.sample_rate = sample_rate
        self.mic_spacing = mic_spacing
        self.c = 343.0

    def steer_beam(self, audio_chunk: np.ndarray, target_azimuth_deg: float) -> np.ndarray:
        """
        Steer beamformer to focus on target azimuth.
        
        audio_chunk: shape (chunk_size, n_channels)
        target_azimuth_deg: target angle in degrees
        
        Returns: beamformed mono output (focused on target)
        """
        target_rad = np.radians(target_azimuth_deg)

        # Compute delays for each channel to align with target
        delays = []
        for ch in range(self.n_channels):
            distance_diff = ch * self.mic_spacing * np.sin(target_rad)
            time_delay = distance_diff / self.c
            sample_delay = time_delay * self.sample_rate
            delays.append(sample_delay)

        # Normalize delays (make first channel delay = 0)
        delays = np.array(delays)
        delays -= delays[0]

        # Apply delays (fractional delay via interpolation)
        output = np.zeros(audio_chunk.shape[0])
        for ch in range(self.n_channels):
            delayed = self.fractional_delay(audio_chunk[:, ch], delays[ch])
            output += delayed

        return output / self.n_channels

    def fractional_delay(self, signal: np.ndarray, delay_samples: float) -> np.ndarray:
        """Apply fractional sample delay via linear interpolation."""
        if abs(delay_samples) < 1e-6:
            return signal

        integer_delay = int(np.floor(delay_samples))
        frac_delay = delay_samples - integer_delay

        # Pad signal for delay
        delayed = np.zeros_like(signal)
        if integer_delay >= 0:
            delayed[integer_delay:] = signal[:-integer_delay] if integer_delay > 0 else signal
        else:
            delayed[:integer_delay] = signal[-integer_delay:]

        # Apply fractional delay via linear interpolation
        for i in range(len(delayed)):
            if i > 0:
                delayed[i] = delayed[i] * (1 - frac_delay) + delayed[i-1] * frac_delay

        return delayed
```

## ⚠️ Difficulties

| Issue | Cause | Fix |
|---|---|---|
| Azimuth noisy | Low SNR, background noise | Use GCC-PHAT, increase beamwidth |
| Multipath reflections | Office walls | Add absorption materials |
| Distance estimates unreliable | Varies with speaker + voice | Use only for rough "near/far" |
| Latency > 100ms | USB buffer bloat | Use ASIO, reduce buffer size |
| Mic impedance mismatch | Different mic models | Calibrate RMS levels per channel |
| 180° ambiguity | Linear array can't distinguish front/back | Add vertical mic or accelerometer |
